開啟 web 伺服器

web伺服器提供網址

執行網頁伺服器

const http = require("http");
http.createServer(function(request, response) {
 console.log(request.url)
   console.log("接收到網頁請求！");
  response.writeHead(200, {"Content-Type": "text/plain"});
  response.write("Hello World");
  response.end();
}).listen(3000);
console.log("Server已開啟port: 3000.");
  • 127.0.0.1:3000 – 內部主機測試網址

node.js 內建語法與模組

  • http 為 Node.js 預設模組
  • http模組提供的函數：createServer，
    預設會回傳一個物件，物件裡有 listen 的方法，可
    指定此 HTTP 伺服器監聽的 Port。
  • request(請求)：客戶端傳送資料給伺服器端
    response(回應):伺服器端傳送資料給客戶端
var http = require("http");


//  node.js 載入模組


http.createServer(function(request, response) {
  console.log(request.url);
  if(request.url == '/'){
    // 路由
    // 資料庫
    console.log("接收到網頁請求！");
    response.writeHead(200, {"Content-Type": "text/HTML"});
    response.write("<h1>index</h1>");
    response.end();
  } else if(request.url == '/index'){
    console.log("接收到網頁請求！");
    response.writeHead(200, {"Content-Type": "text/HTML"});
    response.write("<h1>this is index</h1>");
    response.end();
  } else if(request.url == '/search'){
    console.log("接收到網頁請求！");
    response.writeHead(200, {"Content-Type": "text/HTML"});
    response.write("<h1>search</h1>");
    response.end();
  } else {
    response.writeHead(200, {"Content-Type": "text/HTML"});
    response.write('<head><meta charset="utf-8"></head>');
    response.write("<h1>目前無法顯示頁面</h1>");
    response.end();
  }
  
}).listen(process.env.PORT ||3000);
console.log("Server已開啟port: 3000.");

{
  "name": "server",

  "scripts": {
    "start": "node app.js"
  },

  "license": "ISC"

Express Web 框架環境建立

node.js 市占率最高 web框架，後端應用。

  • 安裝套件指令：npm install
  • 開啟應用程式：npm start

npm install express-generator -g

express 是什麼？
