JS

var hamburgerPrice = 50;
var friesPrice = 40;

document.querySelector('.send').addEventListener("click",function(){
    var hamburgerNum = parseInt(document.querySelector(".hamburgerDOM").value);
    var friesNum = parseInt(document.querySelector(".friesDOM").value);
    var total = hamburgerNum * hamburgerPrice + friesNum * friesPrice;
    document.querySelector(".response").textContent = total;
})

函式 function
指令的意思

程式語言被發明，為了解決人工重複性質的工作

// 宣告一個函式
// 兩個數字加法計算機
// 參數
function sum(num1,num2) {
  return num1 + num2;
}
console.log(sum(1,1));
console.log(sum(2,2));

// 找出變因
// 那些東西是常常變來變去的
// 漢堡幾個、薯條幾個 數量

var VIPTotal = 200;
var hamburgerPrice = 50;
var friesPrice = 30;

function isVIPFun(myName,hamNum, friesNum){
  var total = (hamNum * hamburgerPrice + friesNum * friesPrice);
  var checkVIP = total >= VIPTotal;
  return myName+checkVIP;
}
console.log(isVIPFun("Bob:",1,1));
console.log(isVIPFun("Mary:",10,10));
